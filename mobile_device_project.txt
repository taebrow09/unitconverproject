<script>

let units = {
    length: ["meters","kilometers","feet","miles"],
    weight: ["grams","kilograms","pounds"],
    temperature: ["celsius","fahrenheit"]
};

function updateUnits() {
    let category = $("#category").val();

    $("#fromUnit").html("");
    $("#toUnit").html("");

    let unitList = units[category];

    for (let i = 0; i < unitList.length; i++) {
        let unit = unitList[i];
        $("#fromUnit").append("<option>" + unit + "</option>");
        $("#toUnit").append("<option>" + unit + "</option>");
    }
}

updateUnits();

$("#category").change(function() {
    updateUnits();
});

$("#convertBtn").click(function() {

    let value = parseFloat($("#inputValue").val());
    let from = $("#fromUnit").val();
    let to = $("#toUnit").val();
    let category = $("#category").val();
    let result = 0;

    if (isNaN(value)) {
        $("#result").text("please enter a number!");
        return;
    }

// length
    if (category == "length") {

        if (from == "kilometers") {
            value = value * 1000;
        }
        if (from == "feet") {
            value = value * 0.3048;
        }
        if (from == "miles") {
            value = value * 1609.34;
        }

        if (to == "kilometers") {
            result = value / 1000;
        }
        else if (to == "feet") {
            result = value / 0.3048;
        }
        else if (to == "miles") {
            result = value / 1609.34;
        }
        else {
            result = value;
        }
    }

// weight
    if (category == "weight") {

        if (from == "kilograms") {
            value = value * 1000;
        }
        if (from == "pounds") {
            value = value * 453.592;
        }

        if (to == "kilograms") {
            result = value / 1000;
        }
        else if (to == "pounds") {
            result = value / 453.592;
        }
        else {
            result = value;
        }
    }

// temperature
    if (category == "temperature") {

        if (from == "celsius" && to == "fahrenheit") {
            result = (value * 9/5) + 32;
        }
        else if (from == "fahrenheit" && to == "celsius") {
            result = (value - 32) * 5/9;
        }
        else {
            result = value;
        }
    }

    $("#result").text("Result: " + result.toFixed(2));
});

</script>
